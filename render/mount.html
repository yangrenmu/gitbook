挂载

挂载

以src/platform/web/entry-runtime-with-compiler.js中$mount的实现为例。

/**
 * mount 暂存 Vue原型上的 mount，然后改写 mount
 * 改写后的 mount 作用是拿到由模板编译后的 render 函数，没有下划线
 */
const mount = Vue.prototype.$mount
Vue.prototype.$mount = function(
  el?: string | Element, // 根元素，可以是字符串或者是 DOM 元素
  hydrating?: boolean // 服务端渲染相关，服务端渲染时为 true
): Component {
  // el 是字符串时，将其转为 DOM 元素
  el = el && query(el)

  /* istanbul ignore if */
  // 提示错误，不能使用 body/html 作为根元素，因为根元素会被替换掉
  if (el === document.body || el === document.documentElement) {
    process.env.NODE_ENV !== "production" &&
      warn(
        `Do not mount Vue to <html> or <body> - mount to normal elements instead.`
      )
    return this
  }

  const options = this.$options
  // resolve template/el and convert to render function
  // 没有 render 方法时，获取 template，（template 会被编译成 render 函数）
  if (!options.render) {
    // 先获取模板
    let template = options.template
    if (template) {
      if (typeof template === "string") {
        if (template.charAt(0) === "#") {
          template = idToTemplate(template)
          /* istanbul ignore if */
          if (process.env.NODE_ENV !== "production" && !template) {
            warn(
              `Template element not found or is empty: ${options.template}`,
              this
            )
          }
        }
      } else if (template.nodeType) {
        template = template.innerHTML
      } else {
        if (process.env.NODE_ENV !== "production") {
          warn("invalid template option:" + template, this)
        }
        return this
      }
    } else if (el) {
      // 没有模板时，根元素的 outerHTML 会被当做模板
      template = getOuterHTML(el)
    }
    if (template) {
      /* istanbul ignore if */
      if (process.env.NODE_ENV !== "production" && config.performance && mark) {
        mark("compile")
      }
      // render 方法会生成 vnode, template => render => vnode
      const { render, staticRenderFns } = compileToFunctions(
        template,
        {
          outputSourceRange: process.env.NODE_ENV !== "production",
          shouldDecodeNewlines,
          shouldDecodeNewlinesForHref,
          delimiters: options.delimiters,
          comments: options.comments
        },
        this
      )
      // 将 render 方法添加到 this.$options 上
      options.render = render
      options.staticRenderFns = staticRenderFns

      /* istanbul ignore if */
      if (process.env.NODE_ENV !== "production" && config.performance && mark) {
        mark("compile end")
        measure(`vue ${this._name} compile`, "compile", "compile end")
      }
    }
  }
  // 调用保存的 mount 函数，此时已获得由模板编译过来的 render 函数
  return mount.call(this, el, hydrating)
}

总结：先保存 Vue.prototype.$mount，然后重写原型上的 mount

重写后的 mount

  • 对根节点 el 做了限制，不能挂载到 body、html 上。

  • 如果用户没有自定义 render 方法，vue 会把 el 或者 模板转换成 render 方法。

  • 然后调用之前的 mount 方法。

原先的 mount

最开始原型上的 mount 定义在src/platform/web/runtime/index.js, 它会调用 mountComponent 方法。

//  挂载组件，执行时，this 中已获得由模板编译过来的 render 函数
Vue.prototype.$mount = function(
  el?: string | Element, // 挂载的根元素，可以是字符串或者 DOM，在调用 query 方法时，字符串会被转为DOM对象
  hydrating?: boolean // 服务端渲染时用到
): Component {
  el = el && inBrowser ? query(el) : undefined
  return mountComponent(this, el, hydrating)
}

mountComponent 方法定义在src/core/instance/lifecycle.js中。

export function mountComponent(
  vm: Component, // 组件实例
  el: ?Element, // 挂载的元素
  hydrating?: boolean // 服务端渲染相关
): Component {
  // 组件的根元素
  vm.$el = el
  if (!vm.$options.render) {
    // 如果不存在 render 方法，会创建一个空的 vnode
    vm.$options.render = createEmptyVNode
    // ...很多代码
  }
  // 触发 beforeMount 生命周期
  callHook(vm, "beforeMount")

  let updateComponent

  updateComponent = () => {
    // vm._render()，生成 vnode，在 instance/render.js 中
    // vm._update()，更新 dom
    vm._update(vm._render(), hydrating)
  }

  // 实例化一个 watcher，会调用 updateComponent，先生成 vnode ，然后调用 update 更新 dom；
  // watcher 的另一个作用是：监听数据的变化
  new Watcher(
    vm,
    updateComponent,
    noop,
    {
      before() {
        if (vm._isMounted && !vm._isDestroyed) {
          callHook(vm, "beforeUpdate")
        }
      }
    },
    true /* isRenderWatcher */
  )
  hydrating = false

  // manually mounted instance, call mounted on self
  // mounted is called for render-created child components in its inserted hook
  // 如果挂载完了，_isMounted 置为true，同时触发mounted钩子
  if (vm.$vnode == null) {
    vm._isMounted = true
    callHook(vm, "mounted")
  }
  return vm
}

_render()定义在src/core/instance/render.js中，

// 被模板编译的 render 函数调用
vm._c = (a, b, c, d) => createElement(vm, a, b, c, d, false)
// 被用户手写的 render 方法调用
vm.$createElement = (a, b, c, d) => createElement(vm, a, b, c, d, true)

// 把实例渲染成 vnode
Vue.prototype._render = function(): VNode {
  const vm: Component = this
  // render 方法是由 template/el 编译过来的
  const { render, _parentVnode } = vm.$options

  let vnode
  try {
    // _renderProxy 生产环境是 vm，开发环境是 vm 的 proxy 对象
    // vm.$createElement：用户手写 render 时使用，否则使用 由模板编译过来的 render（render 中有vm._c）
    // render() => vnode
    vnode = render.call(vm._renderProxy, vm.$createElement)
  } catch (e) {
    // ...错误处理
  }
  // ...很多代码
  return vnode
}
